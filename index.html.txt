        class Particle {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.color = color;
                const angle = Math.random() * Math.PI * 2;
                const speed = Math.random() * 8 + 3;
                this.vx = Math.cos(angle) * speed;
                this.vy = Math.sin(angle) * speed;
                this.life = 1;
                this.decay = Math.random() * 0.02 + 0.015;
                this.size = Math.random() * 6 + 3;
            }
            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.vy += 0.06;
                this.life -= this.decay;
                this.size *= 0.99;
            }
            draw() {
                ctx.save();
                ctx.globalAlpha = this.life;
                ctx.shadowBlur = 20;
                ctx.shadowColor = this.color;
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.restore();
            }
        }

        function spawnFirework() {
            if (!document.getElementById('proposal').getBoundingClientRect().top < window.innerHeight) return;

            const y = Math.random() * H * 0.4 + H * 0.1;

                if (p.life <= 0 || p.size < 0.5) {
                    fireworks.splice(i, 1);
                }
            }
            requestAnimationFrame(animateFireworks);

        // ============================================================
        //  AUDIO TOGGLE — Background Music (Web Audio Oscillator)
        // ============================================================

        let audioCtx = null;
        let oscillator = null;
        let gainNode = null;
        let isAudioPlaying = false;

        function initAudio() {
            if (audioCtx) return;
            try {
                audioCtx = new(window.AudioContext || window.webkitAudioContext)();
                const now = audioCtx.currentTime;

                oscillator = audioCtx.createOscillator();
                oscillator.type = 'sine';
                oscillator.frequency.value = 220;

                gainNode = audioCtx.createGain();
                gainNode.gain.value = 0.08;

                const osc2 = audioCtx.createOscillator();
                osc2.type = 'sine';
                osc2.frequency.value = 440;

                const gain2 = audioCtx.createGain();
                gain2.gain.value = 0.04;

                osc2.connect(gain2);
                gain2.connect(gainNode);

                oscillator.connect(gainNode);
                gainNode.connect(audioCtx.destination);

                const lfo = audioCtx.createOscillator();
                lfo.frequency.value = 0.5;
                const lfoGain = audioCtx.createGain();
                lfoGain.gain.value = 4;
                lfo.connect(lfoGain);
                lfoGain.connect(oscillator.frequency);

                oscillator.start(now);
                osc2.start(now);
                lfo.start(now);

                oscillator._osc2 = osc2;
                oscillator._gain2 = gain2;
                oscillator._lfo = lfo;
                oscillator._lfoGain = lfoGain;

                isAudioPlaying = true;
                document.getElementById('audioToggle').classList.remove('muted');
                document.getElementById('audioToggle').querySelector('i').className = 'fas fa-music';
            } catch (e) {
                console.log('Audio could not be initialized.');
            }
        }

        function toggleAudio() {
            if (!audioCtx) {
                initAudio();
                return;
            }
            if (isAudioPlaying) {
                gainNode.gain.value = 0;
                isAudioPlaying = false;
                document.getElementById('audioToggle').classList.add('muted');
                document.getElementById('audioToggle').querySelector('i').className = 'fas fa-volume-mute';
            } else {
                gainNode.gain.value = 0.08;
                isAudioPlaying = true;
                document.getElementById('audioToggle').classList.remove('muted');
                document.getElementById('audioToggle').querySelector('i').className = 'fas fa-music';
                if (audioCtx.state === 'suspended') {
                    audioCtx.resume();
                }
            }
        }

        document.getElementById('audioToggle').addEventListener('click', toggleAudio);

        document.addEventListener('click', function initOnFirstClick() {
            if (!audioCtx) {
                initAudio();
                if (gainNode) gainNode.gain.value = 0.05;
                isAudioPlaying = true;
                document.getElementById('audioToggle').classList.remove('muted');
                document.getElementById('audioToggle').querySelector('i').className = 'fas fa-music';
            }
            document.removeEventListener('click', initOnFirstClick);
        }, { once: true });

        let scrollInitAttempted = false;
        window.addEventListener('scroll', function() {
            if (!audioCtx && !scrollInitAttempted) {
                scrollInitAttempted = true;
                initAudio();
                if (gainNode) gainNode.gain.value = 0.05;
                isAudioPlaying = true;
                document.getElementById('audioToggle').classList.remove('muted');
                document.getElementById('audioToggle').querySelector('i').className = 'fas fa-music';
            }
        }, { once: true });

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = '1';
                    entry.target.style.transform = 'translateY(0)';
                }
            });
        }, { threshold: 0.1 });

                    sec.style.opacity = '1';
                    sec.style.transform = 'translateY(0)';

            }, 200);
